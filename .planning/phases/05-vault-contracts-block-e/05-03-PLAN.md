async syncGlobalToAllAgents(): Promise<{ synced: SyncAgentResult[]; errors: SyncAgentResult[] }> {
  const allAgents = await agentRepo.findAll();
  const synced: SyncAgentResult[] = [];
  const errors: SyncAgentResult[] = [];
  for (const agent of allAgents) {
    if (!agent.directory) {
      errors.push({ slug: agent.slug, error: 'No directory configured — skipped' });
      continue;
    }
    try {
      await this.syncToEnvFile(agent.id, agent.directory, agent.ownerId);
      synced.push({ slug: agent.slug, keys: 0 });
    } catch (err) {
      errors.push({ slug: agent.slug, error: (err as Error).message });
    }
  }
  return { synced, errors };
}

/**
 * Contract test (drift guard) — POST /api/vault/sync-all
 *
 * Ensures `vaultService.syncGlobalToAllAgents()` output conforms to the
 * bridge's `SyncAllResponseSchema` from `@x9-forge/contracts/vault`. If the
 * bridge schema tightens (new required field, stricter validation), this
 * test fails locally before the bridge upgrade ships to Forge prod.
 *
 * Pattern: mock `agentRepo.findAll` + `syncToEnvFile` → call service → parse result via bridge.
 *
 * @see x9-forge-contract-bridge/.planning/phases/05-vault-contracts-block-e/05-RESEARCH.md §"Contract test for sync-all"
 */
import { describe, it, expect, vi, beforeEach, afterEach } from 'vitest';
import { SyncAllResponseSchema } from '@x9-forge/contracts/vault';
import { vaultService } from '../../src/services/vault.service';
import { agentRepo } from '../../src/repositories/agent.repo';

describe('POST /api/vault/sync-all — bridge contract', () => {
  beforeEach(() => {
    vi.restoreAllMocks();
  });

  afterEach(() => {
    vi.restoreAllMocks();
  });

  it('produces a result parseable by bridge SyncAllResponseSchema', async () => {
    // Two agents: one syncs OK, one throws.
    vi.spyOn(agentRepo, 'findAll').mockResolvedValue([
      { id: 1, slug: 'test-agent-a', directory: '/tmp/test-a', ownerId: 7 } as any,
      { id: 2, slug: 'test-agent-b', directory: '/tmp/test-b', ownerId: 7 } as any,
    ]);
    vi.spyOn(vaultService as any, 'syncToEnvFile')
      .mockResolvedValueOnce(undefined)
      .mockRejectedValueOnce(new Error('ENOENT: workspace not found'));

    const result = await vaultService.syncGlobalToAllAgents();

    // Route wraps with ok:true — mirror that shape for parse.
    const parsed = SyncAllResponseSchema.parse({ ok: true, ...result });

    expect(parsed.synced).toHaveLength(1);
    expect(parsed.synced[0]!.slug).toBe('test-agent-a');
    expect(parsed.errors).toHaveLength(1);
    expect(parsed.errors[0]!.slug).toBe('test-agent-b');
    expect(parsed.errors[0]!.error).toContain('ENOENT');
  });

  it('handles zero-agent edge case (empty arrays)', async () => {
    vi.spyOn(agentRepo, 'findAll').mockResolvedValue([]);

    const result = await vaultService.syncGlobalToAllAgents();
    const parsed = SyncAllResponseSchema.parse({ ok: true, ...result });

    expect(parsed.synced).toEqual([]);
    expect(parsed.errors).toEqual([]);
  });

  it('handles "no directory configured" skip case', async () => {
    vi.spyOn(agentRepo, 'findAll').mockResolvedValue([
      { id: 1, slug: 'broken-agent', directory: null, ownerId: 7 } as any,
    ]);

    const result = await vaultService.syncGlobalToAllAgents();
    const parsed = SyncAllResponseSchema.parse({ ok: true, ...result });
